These tests cover operators.

### Add

```
>>> 7 + 2
9

>>> 7 + [2, 3]
[9, 10]

>>> [7, 8] + 2
[9, 10]

>>> [7, 8] + [2, 3]
[9, 11]

>>> 7 + nil
nil

>>> 7. + 2.
9.0

>>> 7. + [2., 3.]
[9.0, 10.0]

>>> [7., 8.] + 2.
[9.0, 10.0]

>>> [7., 8.] + [2., 3.]
[9.0, 11.0]

>>> 7. + nan
nan

```

### Subtract

```
>>> 7 - 2
5

>>> 7 - [2, 3]
[5, 4]

>>> [7, 8] - 2
[5, 6]

>>> [7, 8] - [2, 3]
[5, 5]

>>> 7 - nil
nil

>>> 7. - 2.
5.0

>>> 7. - [2., 3.]
[5.0, 4.0]

>>> [7., 8.] - 2.
[5.0, 6.0]

>>> [7., 8.] - [2., 3.]
[5.0, 5.0]

>>> 7. - nan
nan

```

### Multiply

```
>>> 7 * 2
14

>>> 7 * [2, 3]
[14, 21]

>>> [7, 8] * 2
[14, 16]

>>> [7, 8] * [2, 3]
[14, 24]

>>> 7 * nil
nil

>>> 7. * 2.
14.0

>>> 7. * [2., 3.]
[14.0, 21.0]

>>> [7., 8.] * 2.
[14.0, 16.0]

>>> [7., 8.] * [2., 3.]
[14.0, 24.0]

>>> 7. * nan
nan

```

### Divide

```
>>> 7 / 2
3

>>> 7 / [2, 3]
[3, 2]

>>> [7, 8] / 2
[3, 4]

>>> [7, 8] / [2, 3]
[3, 2]

>>> 7 / nil
nil

>>> 7. / 2.
3.5

>>> 7. / [2., 3.]
[3.5, 2.333333]

>>> [7., 8.] / 2.
[3.5, 4.0]

>>> [7., 8.] / [2., 3.]
[3.5, 2.666667]

>>> 7. / nan
nan

```

### Modulo

```
>>> 7 % 2
1

>>> 7 % [2, 3]
[1, 1]

>>> [7, 8] % 2
[1, 0]

>>> [7, 8] % [2, 3]
[1, 2]

>>> 7 % nil
nil

```

### Left Shift

```
>>> 7 << 2
28

>>> 7 << [2, 3]
[28, 56]

>>> [7, 8] << 2
[28, 32]

>>> [7, 8] << [2, 3]
[28, 64]

>>> 7 << nil
nil

```

### Right Shift

```
>>> 7 >> 2
1

>>> 7 >> [2, 3]
[1, 0]

>>> [7, 8] >> 2
[1, 2]

>>> [7, 8] >> [2, 3]
[1, 1]

>>> 7 >> nil
nil

```

### Bitwise Or

```
>>> 7 | 2
7

>>> 7 | [2, 3]
[7, 7]

>>> [7, 8] | 2
[7, 10]

>>> [7, 8] | [2, 3]
[7, 11]

>>> 7 | nil
nil

```

### Bitwise And

```
>>> 7 & 2
2

>>> 7 & [2, 3]
[2, 3]

>>> [7, 8] & 2
[2, 0]

>>> [7, 8] & [2, 3]
[2, 0]

>>> 7 & nil
nil

```
